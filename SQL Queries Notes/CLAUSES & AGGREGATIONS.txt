SELECT name , salary FROM employees WHERE salary > 100000; -- WHERE CLAUSE USE TO FILTER THE ROWS

SELECT * FROM employees ORDER BY salary ASC;

SELECT COUNT(*) as Count_of_employees , role ,
SUM(salary) , AVG(salary) FROM employees GROUP BY role ; 

SELECT COUNT(*) as rich_employees , role , SUM(salary) FROM employees 
 GROUP BY role HAVING SUM(salary) > 170000 ;


-- **** AGGREGATING FUNCTIONS **** 

SELECT COUNT(*) as total_employees FROM employees;

SELECT SUM(salary) as total_salary FROM employees; -- salary / num_of_emp

SELECT AVG(salary) as avg_salary FROM employees;

SELECT MAX(salary) as max_salary FROM employees;  -- 123456

SELECT MIN(salary) as max_salary FROM employees;  -- 80000

SELECT name , salary FROM employees WHERE salary = 
(SELECT MIN(salary) as max_salary FROM employees) ;
