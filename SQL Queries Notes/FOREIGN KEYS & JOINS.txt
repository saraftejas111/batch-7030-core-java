-- JOINS
SELECT * FROM subject;
SELECT * FROM faculty;

-- INNER JOIN : returns only the matching columns data

SELECT subject.subId , subject.subjectName , subject.duratioMonths , 
faculty.id ,  faculty.name , faculty.subCode FROM subject 
INNER JOIN faculty ON subject.subId =  faculty.subCode ; 

-- LEFT JOIN : returns all the data of left side table only & declares right side table columns null if  not mathced

SELECT subject.subId , subject.subjectName , subject.duratioMonths , 
faculty.id ,  faculty.name , faculty.subCode FROM subject 
LEFT JOIN faculty ON subject.subId =  faculty.subCode ; 

-- RIGHT JOIN : returns all the data of RIGHT side table only & declares LEFT side table columns null if not mathced

SELECT subject.subId , subject.subjectName , subject.duratioMonths , 
faculty.id ,  faculty.name , faculty.subCode FROM subject 
RIGHT JOIN faculty ON subject.subId =  faculty.subCode ; 

-- OUTER JOIN : returns all the data of left & right side table columns.. null if not mathced

SELECT subject.subId , subject.subjectName , subject.duratioMonths , 
faculty.id ,  faculty.name , faculty.subCode FROM subject 
LEFT JOIN faculty ON subject.subId =  faculty.subCode
